//1.穷举, o(n^2)
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<int> re;
        int i = 0, j = 0;
        int n = nums.size();
        for (i = 0; i < n; i++)
        {
            for (j = i + 1; j < n; j++)
            {
                if (nums[i] + nums[j] == target)
                {
                    re.push_back(i);
                    re.push_back(j);
                    return re;
                }
            }
        }
        return re;
    }
};

//2.两遍哈希
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        map<int, int> vToM;//从值到下标的映射
        for (int i = 0; i < nums.size(); i++) vToM.insert(map<int, int>::value_type(nums[i], i));
        vector<int> res;
        for (int i = 0; i < nums.size(); i++)
        {
            if (vToM.count(target - nums[i]) && vToM[target - nums[i]] != i)
            {
                res.push_back(i);
                res.push_back(vToM[target - nums[i]]);
                break;
            }
        }
        return res;
    }
};

//3.一遍哈希
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        map<int, int> vToM;
        vector<int> res;
        for (int i = 0; i < nums.size(); i++)
        {
            if (vToM.count(target - nums[i]))//先找再放表里
            {
                res.push_back(vToM[target - nums[i]]);
                res.push_back(i);
                break;
            }
            vToM.insert(map<int, int>::value_type(nums[i], i));
        }
        return res;
    }
};
